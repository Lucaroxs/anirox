API Documentation

GET /search

Search for anime by name

Parameters

q required

The anime name to search for

Example Request

GET https://api.animeapi.dev/search?q=attack%20on%20titan

Response Schema

{
  "success": boolean,
  "data": [
    {
      "id": integer,
      "name": string,
      "name_english": string,
      "description": string,
      "poster": string,
      "backdrop": string,
      "year": integer,
      "episode_count": integer,
      "genres": [string],
      "rating": number
    }
  ]
}
GET /episodes

Get episode video links and information

Parameters

titleId required

The anime ID from search results

season required

Season number (default: 1)

episode required

Episode number

Response Schema

{
  "success": boolean,
  "data": {
    "episode_info": {
      "name": string,
      "description": string,
      "poster": string
    },
    "sources": {
      "provider_name": [
        {
          "id": string,
          "url": string,
          "quality": string,
          "language": string
        }
      ]
    }
  }
}
GET /anime-info

Get detailed information about a specific anime

Parameters

titleId required

The anime ID from search results

Example Request

GET https://api.animeapi.dev/anime-info?titleId=1735

Response Schema

{
  "success": boolean,
  "data": {
    "message": string,
    "url": string,
    "note": string
  }
}

Code Examples

// Using fetch API
async function searchAnime(query) {
  try {
    const response = await fetch(
      `https://api.animeapi.dev/search?q=${encodeURIComponent(query)}`
    );
    
    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }
    
    const data = await response.json();
    return data;
    
  } catch (error) {
    console.error('Error searching anime:', error);
    throw error;
  }
}

// Usage
searchAnime('one piece')
  .then(data => {
    console.log('Found anime:', data.data);
  })
  .catch(error => {
    console.error('Search failed:', error);
  });
